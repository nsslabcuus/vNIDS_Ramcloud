593 template <typename T, typename A>
594 T *HashContainer<T, A>::set(iterator &it, T *element, bool balance)
595 {
596  click_hash_assert(it._hc == this && it._bucket < _rep.nbuckets);
597  click_hash_assert(bucket(_rep.hashkey(element)) == it._bucket);
598  click_hash_assert(!it._element || _rep.hashkeyeq(_rep.hashkey(element), _rep.hashkey(it._element)));
599  T *old = it.get();
600  if (unlikely(old == element))
601  return old;
602  if (!element) {
603  --_rep.size;
604  if (!(*it._pprev = it._element = _rep.hashnext(old)))
605  ++it;
606  return old;
607  }
608  if (old)
609  _rep.hashnext(element) = _rep.hashnext(old);
610  else {
611  ++_rep.size;
612  if (unlikely(unbalanced()) && balance) {
613  rehash(bucket_count() + 1);
614  it._bucket = bucket(_rep.hashkey(element));
615  it._pprev = &_rep.buckets[it._bucket];
616  }
617  if (!(_rep.hashnext(element) = *it._pprev))
618  _rep.first_bucket = 0;
619  }
620  *it._pprev = it._element = element;
621  return old;
622 }

673 template <typename T, typename A>
674 void HashContainer<T, A>::rehash(bucket_count_type n)
675 {
676  bucket_count_type new_nbuckets = 1;
677  while (new_nbuckets < n && new_nbuckets < max_bucket_count)
678  new_nbuckets = ((new_nbuckets + 1) << 1) - 1;
679  click_hash_assert(new_nbuckets > 0 && new_nbuckets <= max_bucket_count);
680  if (_rep.nbuckets == new_nbuckets)
681  return;
682 
683  T **new_buckets = (T **) CLICK_LALLOC(sizeof(T *) * new_nbuckets);
684  for (bucket_count_type b = 0; b < new_nbuckets; ++b)
685  new_buckets[b] = 0;
686 
687  bucket_count_type old_nbuckets = _rep.nbuckets;
688  T **old_buckets = _rep.buckets;
689  _rep.nbuckets = new_nbuckets;
690  _rep.buckets = new_buckets;
691  _rep.first_bucket = 0;
692  _rep.bucket_divider = libdivide_u32_gen(_rep.nbuckets);
693  click_hash_assert(_rep.nbuckets == libdivide_u32_recover(&_rep.bucket_divider));
694 
695  for (size_t b = 0; b < old_nbuckets; b++)
696  for (T *element = old_buckets[b]; element; ) {
697  T *next = _rep.hashnext(element);
698  bucket_count_type new_b = bucket(_rep.hashkey(element));
699  _rep.hashnext(element) = new_buckets[new_b];
700  new_buckets[new_b] = element;
701  element = next;
702  }
703 
704  CLICK_LFREE(old_buckets, sizeof(T *) * old_nbuckets);
705 }
