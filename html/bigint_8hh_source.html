146  static limb_type inverse(limb_type x) {
147  limb_type y1 = ~x, y0 = ~limb_zero;
148 
149  // now calculate [y1y0] / x
150  limb_type x1 = high(x), x0 = low(x);
151  limb_type q1 = y1 / x1;
152  limb_type r1 = y1 - q1 * x1;
153  limb_type m = q1 * x0;
154  r1 = r1 * limb_half | high(y0);
155  if (r1 < m) {
156  --q1, r1 += x;
157  if (r1 >= x && r1 < m)
158  --q1, r1 += x;
159  }
160  r1 -= m;
161 
162  limb_type q0 = r1 / x1;
163  limb_type r0 = r1 - q0 * x1;
164  m = q0 * x0;
165  r0 = r0 * limb_half | low(y0);
166  if (r0 < m) {
167  --q0, r0 += x;
168  if (r0 >= x && r0 < m)
169  --q0, r0 += x;
170  }

274  static limb_type divide(limb_type *x, const limb_type *a, int n,
275  limb_type b) {
276  x += n - 1; /* Make x point at most significant quotient limb */
277  a += n - 1; /* Make a point at most significant dividend limb */
278  limb_type r = 0;
279 
280  if (b & limb_high_bit) {
281  /* High quotient limb is 0 or 1, skip a divide step. */
282  r = *a;
283  *x = (r >= b);
284  r -= (b & -*x);
285  --x, --a, --n;
286 
287  /* Multiply-by-inverse, divisor already normalized. */
288  limb_type b_inverse = inverse(b);
289  while (n > 0) {
290  preinverted_divide(*x, r, r, *a, b, b_inverse);
291  --x, --a, --n;
292  }
293 
294  } else {
295  /* Most significant bit of divisor == 0. */
296  /* Skip a division if high < divisor (high quotient 0). Testing
297  here before normalizing will still skip as often as
298  possible. */
299  if (*a < b) {
300  r = *a;
301  *x = 0;
302  --x, --a, --n;
303  }
304 
305  if (n != 0) {
306  int norm = ffs_msb(b) - 1; // number of most significant 0 bits
307  b <<= norm;
308  r <<= norm;
309  limb_type b_inverse = inverse(b);
310 
311  limb_type a1 = *a;
312  r |= (a1 >> (limb_bits - norm));
313  --a, --n;
314  while (n > 0) {
315  limb_type a0 = *a;
316  preinverted_divide(*x, r, r, ((a1 << norm) | (a0 >> (limb_bits - norm))), b, b_inverse);
317  --x, --a, --n;
318  a1 = a0;
319  }
320  preinverted_divide(*x, r, r, a1 << norm, b, b_inverse);
321  r >>= norm;
322  }
323  }
324 
325  return r;
326  }

335  // need d chars, min d s.t. 10^d >= 2^(sizeof(limb_type) * 8 * n)

362  /* Divide the two-limb number in (a1,a0) by b, with b_inverse being the
363  largest limb not larger than (2**(2*limb_bits))/b - (2**limb_bits). If
364  this would yield overflow, b_inverse should be the largest possible
365  number (i.e., only ones). For correct operation, the most significant
366  bit of b has to be set. Put the quotient in q and the remainder in
367  r. */
368  /* Like udiv_qrnnd_preinv, but branch-free. */
369  static void preinverted_divide(limb_type &q, limb_type &r,
370  limb_type a1, limb_type a0, limb_type b,
371  limb_type b_inverse) {
372  limb_type a0_mask = (a0 & limb_high_bit ? ~limb_zero : limb_zero);
373  limb_type a0_adjusted = a0 + (b & a0_mask);
374  limb_type x1, x0;
375  multiply(x1, x0, b_inverse, a1 - a0_mask);
376  add(x1, x0, x1, x0, a1, a0_adjusted);
377  limb_type q1 = ~x1;
378  multiply(x1, x0, q1, b);
379  add(x1, x0, x1, x0, a1, a0);
380  x1 -= b;
381  r = x0 + (b & x1);
382  q = x1 - q1;
383  }
