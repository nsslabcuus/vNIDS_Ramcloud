181 inline bool
182 GapRate::need_update(const Timestamp &now)
183 {
184  // this is an approximation of:
185  // unsigned need = (unsigned) ((now.usec() / 1000000.0) * _rate)
186  unsigned need = (now.usec() << UGAP_SHIFT) / _ugap;
187 
188  if (_tv_sec < 0) {
189  // 27.Feb.2005: often OK to send a packet after reset unless rate is
190  // 0 -- requested by Bart Braem
191  // check include/click/gaprate.hh (1.2)
192  _tv_sec = now.sec();
193  _sec_count = need + ((now.usec() << UGAP_SHIFT) - (need * _ugap) > _ugap / 2);
194  } else if (now.sec() > _tv_sec) {
195  _tv_sec = now.sec();
196  if (_sec_count > 0)
197  _sec_count -= _rate;
198  }
199 
200 #if DEBUG_GAPRATE
201  click_chatter("%p{timestamp} -> %u @ %u [%d]", &now, need, _sec_count, (int)need >= _sec_count);
202 #endif
203  return ((int)need >= _sec_count);
204 }
